package service

import (
	"context"
	"gateway-front-admin/models"
	"gateway-front-admin/restapi/operations/admin_gateway_service"
	"github.com/{{ .PackageStruct.GitCompanyName}}/{{ .PackageStruct.GitRepositoryName}}"
	"github.com/getsentry/sentry-go"
	"github.com/go-openapi/runtime/middleware"
	"github.com/go-openapi/strfmt"
	"google.golang.org/grpc/codes"
	"google.golang.org/grpc/status"
)

func (s *Service) List{{ .Name}}() admin_gateway_service.List{{ .Name}}HandlerFunc {
	return func(params admin_gateway_service.List{{ .Name}}Params) middleware.Responder {
		responder := admin_gateway_service.NewList{{ .Name}}OK()

		response, err := s.list{{ .Name}}(params)
		if err != nil {
			sentry.CaptureException(err)
		}

		return responder.WithPayload(response)
	}
}

func (s *Service) list{{ .Name}}(params admin_gateway_service.List{{ .Name}}Params) (*models.List{{ .Name}}Response, error) {
	response := &models.List{{ .Name}}Response{}
	response.Status = models.StatusOK

	info, ok := s.GetTokenInfo(params.HTTPRequest.Header.Get(xAuthToken))
	if !ok {
		response.Status = models.StatusBADCREDENTIALS
		return response, nil
	}

	if !SliceIsIntersect(info.UserRoles, []string{roleAdmin}) {
		response.Status = models.StatusBADCREDENTIALS
		return response, nil
	}

	request := {{ .PackageStruct.PackageName}}.List{{ .Name}}Request{
		Limit: params.Limit,
	}

	if params.Offset != nil {
		request.Offset = *params.Offset
	}

    // TODO Допиши параметры

	items, err := s.{{ .Name}}Repository.List{{ .Name}}(context.Background(), &request)
	if err != nil && status.Code(err) != codes.NotFound {
		response.Status = models.StatusINTERNALERROR
		return response, err
	}

	if err != nil && status.Code(err) == codes.NotFound {
		response.Status = models.StatusNOCONTENT
		return response, nil
	}

	response.Payload.Items = make([]*models.{{ .Name}}, 0, len(items.Items))

	response.Meta = models.Meta{
		Total:      items.Total,
		IsLastPage: request.Limit+request.Offset >= items.Total,
	}

	for _, item := range items.Items {
		item := models.{{ .Name}}{
			ID:                  item.Id,
			CreatedAt:           strfmt.DateTime(item.CreatedAt.AsTime()),
			UpdatedAt:           strfmt.DateTime(item.UpdatedAt.AsTime()),
			// TODO Допиши параметры
		}

		response.Payload.Items = append(response.Payload.Items, &item)
	}

	return response, nil
}
